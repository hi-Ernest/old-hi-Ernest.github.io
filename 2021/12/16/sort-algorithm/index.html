public void sort(int nums) &#123;
    if (nums.length <= 1) &#123;
        return;
    &#125;
    quickSort(nums, 0, nums.length - 1)

public void quickSort(int[] nums, int left, int right) &#123;
    if (left >= right) &#123;
        return;
    &#125;
    int partition = getPartition(nums, left, right);
    quickSort(nums, left, partition - 1);
    quickSort(nums, partition + 1, right);
&#125;

public int getPartition(int[] nums, int left, int right) &#123;
    int i = left;
    int sentry = nums[right];
    for (int j = left; j <= right - 1; j++) &#123;
        if (nums[j] < sentry) &#123;
            swap(nums, i, j);
            i++;
        &#125;
    &#125;
    swap(nums, i, right);
    return i;
&#125;

//归并排序
public int[] mergeKNums(int[][] nums) &#123;
    return mergeSort(nums, 0, nums.length - 1);
&#125;

public int[] mergeSort(int[][] nums, int left, int right) &#123;
    if (left == right) &#123;
        return nums[left];
    &#125;
    int mid = left + (right - left) / 2;
    int[] nums1 = mergeSort(nums, left, mid);
    int[] nums2 = mergeSort(nums, mid + 1, right);

//堆排序
public class HeapSort &#123;

    /**
     * 比较 + 交换
     * 建堆的时间复杂度：O(N)
     * 时间复杂度都一样：O(N * logN)
     *
     * 思考与快排有什么区别？哪个好？
     *
     * @param nums 待排序的数组
     * @return 已好排序数组（从小到大）
     */
    public int[] sortArray(int[] nums) &#123;
        heapSort(nums);
        return nums;
    &#125;

    /**
     * 基于二叉堆中最大堆实现（抽象成完全二叉树）
     * 构建大顶堆
     * 依次将最大元素（根节点）与待排序数组最后一个元素进行交换（二叉树最深层最右边的叶子节点）
     * 每次遍历，刷新最后一个元素位置（自减1），直到与首元素相交，即完成排序
     *
     * @param nums 待排序的数组
     */
    public void heapSort(int[] nums) &#123;
        int len = nums.length;
        //构建大顶堆 O(N)
        for (int i = len/2; i >= 0; i--) &#123;
            heapify(nums, len, i);
        &#125;
        for (int i = len - 1; i >= 1; i--) &#123;
            swap(nums, 0, i);
            //i：需要堆化的元素个数 0：下标
            heapify(nums, i, 0);
        &#125;
    &#125;

    //自上而下的堆化
    //时间复杂度：O(N * logN)
    public void heapify(int[] nums, int len, int index) &#123;
        //index 是父节点
        int left = 2 * index + 1;
        int right = 2 * index + 2;
        int maxIndex = index;
        if (left < len && nums[left] > nums[maxIndex]) &#123;
            maxIndex = left;
        &#125;
        if (right < len && nums[right] > nums[maxIndex]) &#123;
            maxIndex = right;
        &#125;
        if (maxIndex != index) &#123;
            swap(nums, index, maxIndex);
            heapify(nums, len, maxIndex);
        &#125;
    &#125;

    public void swap(int[] nums, int i, int j) &#123;
        if (i == j) &#123;
            return;
        &#125;
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
    &#125;
